batteries_tonne = pdata['amount processed (tonne)'].sum()
acquisition = 358 # $/tonne

# total plant cost
pin = pdata.groupby([
    "Storage Limit (%)",
    "Storage Cost ($/tonne/yr)"
]).sum()[
    "total cost ($)"
].reset_index().pivot(
    "Storage Limit (%)",
    "Storage Cost ($/tonne/yr)",
    "total cost ($)",
)


# Transport cost
tin = tdata.groupby([
    "Storage Limit (%)",
    "Storage Cost ($/tonne/yr)"
]).sum()['transportation cost ($)'].reset_index().pivot(
    "Storage Limit (%)",
    "Storage Cost ($/tonne/yr)",
    "transportation cost ($)",
)

# disposal cost
din = ddata.groupby([
